local Base = require "core.Object"
local GameState = class(Base)

function GameState:ctor( name, mgr)
    -- print( self.__name)
    self.name = name 
    self.mgr = mgr
    Base.ctor(self, name, mgr)
end


function GameState:ChangeState(stateName)
    (self.mgr):ChangeState(stateName)
end

return GameState
